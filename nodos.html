Creación y modificación

createElement y createTextNode

createElement Crea un nodo de elemento en el DOM. (representación de una etiqueta HTML como nav, div, p…)
createTextNode Crea un nodo de texto en el DOM. (Un nodo de texto puede contener únicamente una cadena y no puede tener hijos). 


const listas = document.querySelectorAll("ul");

function creaText() {
    let para = document.createElement("p");
    let node = document.createTextNode("Esto es nuevo.");
    para.appendChild(node); 
    const idCreaText = document.getElementById("idCreaText");
    idCreaText.appendChild(para); 
    }

insertBefore

Inserta un nodo antes de otro nodo hijo existente.


function insert() {
    const segundaLista = listas[2];
    const creaLi = document.createElement("li");
    creaLi.textContent = "Nuevo elemento de lista";
            
    segundaLista.insertBefore(creaLi, segundaLista.firstChild); 
    }

removeChild

Elimina un nodo hijo de su nodo padre.


function borra() {
    const terceraLista = listas[3];
    const borrarLi = terceraLista.lastChild;
        
    terceraLista.removeChild(borrarLi);   
    }

cloneNode

Clona un nodo existente.


function clona() {
    const cuartaLista = listas[4];
    const copia = cuartaLista.cloneNode(true);
            
    cuartaLista.appendChild(copia); 
    }
